    function _addMultiple(address[] memory users, uint256 amount, string memory reason) internal {
        for (uint i = 0; i < users.length; i++) {
            userAirdrops[users[i]].push(UserAirdrop({
                amount: amount,
                claimed: false,
                reason: reason
            }));
            isEligible[users[i]] = true;
            totalEligible += amount;
            emit AirdropAdded(users[i], amount, reason);
        }
    }
    
    // ============ ОСНОВНЫЕ ФУНКЦИИ ============
    
    function getAirdrops(address user) external view returns (UserAirdrop[] memory) {
        return userAirdrops[user];
    }
    
    function getTotalAirdropAmount(address user) external view returns (
        uint256 total, 
        uint256 claimed, 
        uint256 pending
    ) {
        UserAirdrop[] storage drops = userAirdrops[user];
        for (uint i = 0; i < drops.length; i++) {
            total += drops[i].amount;
            if (drops[i].claimed) {
                claimed += drops[i].amount;
            } else {
                pending += drops[i].amount;
            }
        }
        
        (uint256 nftPending, uint256 nftClaimed) = _getNFTStats(user);
        pending += nftPending;
        claimed += nftClaimed;
        total += nftPending + nftClaimed;
    }
    
    function _getNFTStats(address user) internal view returns (uint256 pending, uint256 claimed) {
        // DRUB Badge NFT - 15,000 HASH
        if (DRUB_BADGE_NFT.balanceOf(user) > 0) {
            bool alreadyAdded = false;
            for (uint i = 0; i < userAirdrops[user].length; i++) {
                if (keccak256(bytes(userAirdrops[user][i].reason)) == keccak256(bytes("DRUB Badge NFT"))) {
                    alreadyAdded = true;
                    if (!userAirdrops[user][i].claimed) {
                        pending += DRUB_AMOUNT;
                    } else {
                        claimed += DRUB_AMOUNT;
                    }
                    break;
                }
            }
            if (!alreadyAdded) {
                pending += DRUB_AMOUNT;
            }
        }
        
        // Dolphin NFT - 30,000 HASH
        if (DOLPHIN_NFT.balanceOf(user) > 0) {
            bool alreadyAdded = false;
            for (uint i = 0; i < userAirdrops[user].length; i++) {
                if (keccak256(bytes(userAirdrops[user][i].reason)) == keccak256(bytes("Dolphin NFT"))) {
                    alreadyAdded = true;
                    if (!userAirdrops[user][i].claimed) {
                        pending += DOLPHIN_AMOUNT;
                    } else {
                        claimed += DOLPHIN_AMOUNT;
                    }
                    break;
                }
            }
            if (!alreadyAdded) {
                pending += DOLPHIN_AMOUNT;
            }
        }
        
        return (pending, claimed);
    }
    
    function claimAllAirdrops() external {
        uint256 totalAmount = 0;
        
        // Клейм базовых наград
        for (uint i = 0; i < userAirdrops[msg.sender].length; i++) {
            if (!userAirdrops[msg.sender][i].claimed && userAirdrops[msg.sender][i].amount > 0) {
                userAirdrops[msg.sender][i].claimed = true;
                totalAmount += userAirdrops[msg.sender][i].amount;
                emit AirdropClaimed(msg.sender, userAirdrops[msg.sender][i].amount, userAirdrops[msg.sender][i].reason);
            }
        }
        
        // Клейм NFT наград
        totalAmount += _claimAdditionalRewards(msg.sender);
        
        require(totalAmount > 0, "No pending airdrops");
        totalClaimed += totalAmount;
        require(hashToken.transfer(msg.sender, totalAmount), "Transfer failed");
    }
    
    function _claimAdditionalRewards(address user) internal returns (uint256) {
        uint256 amount = 0;
        
        // DRUB Badge NFT - 15,000 HASH
        if (DRUB_BADGE_NFT.balanceOf(user) > 0) {
            bool alreadyAdded = false;
            for (uint i = 0; i < userAirdrops[user].length; i++) {
                if (keccak256(bytes(userAirdrops[user][i].reason)) == keccak256(bytes("DRUB Badge NFT"))) {
                    if (!userAirdrops[user][i].claimed) {
                        userAirdrops[user][i].claimed = true;
                        amount += DRUB_AMOUNT;
                        emit AirdropClaimed(user, DRUB_AMOUNT, "DRUB Badge NFT");
                    }
                    alreadyAdded = true;
                    break;
                }
            }
            if (!alreadyAdded) {
                userAirdrops[user].push(UserAirdrop({
                    amount: DRUB_AMOUNT,
                    claimed: true,
                    reason: "DRUB Badge NFT"
                }));
                amount += DRUB_AMOUNT;
                isEligible[user] = true;
                totalEligible += DRUB_AMOUNT;
                emit AirdropClaimed(user, DRUB_AMOUNT, "DRUB Badge NFT");
            }
        }
        
        // Dolphin NFT - 30,000 HASH
        if (DOLPHIN_NFT.balanceOf(user) > 0) {
            bool alreadyAdded = false;
            for (uint i = 0; i < userAirdrops[user].length; i++) {
                if (keccak256(bytes(userAirdrops[user][i].reason)) == keccak256(bytes("Dolphin NFT"))) {
                    if (!userAirdrops[user][i].claimed) {
                        userAirdrops[user][i].claimed = true;
                        amount += DOLPHIN_AMOUNT;
                        emit AirdropClaimed(user, DOLPHIN_AMOUNT, "Dolphin NFT");
                    }
                    alreadyAdded = true;
                    break;
                }
            }
            if (!alreadyAdded) {
                userAirdrops[user].push(UserAirdrop({
                    amount: DOLPHIN_AMOUNT,
                    claimed: true,
                    reason: "Dolphin NFT"
                }));
                amount += DOLPHIN_AMOUNT;
                isEligible[user] = true;
                totalEligible += DOLPHIN_AMOUNT;
                emit AirdropClaimed(user, DOLPHIN_AMOUNT, "Dolphin NFT");
            }
        }
        
        return amount;
    }

    function getContractBalance() external view returns (uint256) {

    function withdrawTokens(address token, uint256 amount) external onlyOwner {
        if (token == address(0)) {
            (bool success, ) = payable(owner).call{value: amount}("");
            require(success, "ETH transfer failed");
        } else {
            require(IERC20(token).transfer(owner, amount), "Token transfer failed");
        }
    }

    color: "text-yellow-400",
    bg: "bg-yellow-900/20",
    border: "border-yellow-500/30",
    label: "DRUB Badge Holder",
    description: "Owns DRUB Badge NFT",
    category: "NFT"
  },
  "Dolphin NFT": {
    icon: Fish,
    color: "text-cyan-400",
    bg: "bg-cyan-900/20",
    border: "border-cyan-500/30",
    label: "Dolphin NFT Holder",
    description: "Owns Dolphin NFT",
    category: "NFT"
  }
};

interface AirdropInfo {
  amount: string;
  claimed: boolean;
  reason: string;
}

interface Stats {
  total: string;
  claimed: string;
  pending: string;
}

const AirdropScanner: React.FC = () => {
  const [address, setAddress] = useState('');
  const [airdrops, setAirdrops] = useState<AirdropInfo[]>([]);
  const [stats, setStats] = useState<Stats>({ total: '0', claimed: '0', pending: '0' });
  const [loading, setLoading] = useState(false);
  const [scanning, setScanning] = useState(false);
  const [account, setAccount] = useState<string | null>(null);

  const CONTRACT_ADDRESS = '0x...'; // Вставьте адрес вашего контракта
  const ABI = [
    "function getAirdrops(address user) view returns ((uint256 amount, bool claimed, string reason)[])",
    "function getTotalAirdropAmount(address user) view returns (uint256, uint256, uint256)",
    "function claimAllAirdrops()",
    "function hashToken() view returns (address)"
  ];

  // Подключение к MetaMask
  const connectWallet = async () => {
    if (!window.ethereum) {
      alert('Please install MetaMask');
      return;
    }
    try {
      const provider = new ethers.BrowserProvider(window.ethereum);
      const signer = await provider.getSigner();
      const address = await signer.getAddress();
      setAccount(address);
      setAddress(address);
      scanAddress(address);
    } catch (error) {
      console.error(error);
    }
  };

  // Сканирование адреса
  const scanAddress = async (addr: string) => {
    if (!ethers.isAddress(addr)) {
      alert('Invalid address');
      return;
    }

    setScanning(true);
    setLoading(true);

    try {
      const provider = new ethers.BrowserProvider(window.ethereum);
      const contract = new ethers.Contract(CONTRACT_ADDRESS, ABI, provider);

      // Получаем все дропы
      const drops = await contract.getAirdrops(addr);
      const formattedDrops = drops.map((d: any) => ({
        amount: ethers.formatEther(d.amount),
        claimed: d.claimed,
        reason: d.reason
      }));
      setAirdrops(formattedDrops);

      // Получаем статистику
      const statsData = await contract.getTotalAirdropAmount(addr);
      setStats({
        total: ethers.formatEther(statsData[0]),
        claimed: ethers.formatEther(statsData[1]),
        pending: ethers.formatEther(statsData[2])
      });

    } catch (error) {
      console.error(error);
      alert('Error scanning address');
    } finally {
      setLoading(false);
      setScanning(false);
    }
  };

  // Клейм всех наград
  const claimAll = async () => {
    if (!window.ethereum) {
      alert('Please install MetaMask');
      return;
    }

    try {
      const provider = new ethers.BrowserProvider(window.ethereum);
      const signer = await provider.getSigner();
      const contract = new ethers.Contract(CONTRACT_ADDRESS, ABI, signer);

      const tx = await contract.claimAllAirdrops();
      await tx.wait();
      
      alert('🎉 All airdrops claimed successfully!');
      if (address) scanAddress(address);
      
    } catch (error) {
      console.error(error);
      alert('Error claiming airdrops');
    }
  };

  // Получение конфигурации для критерия
  const getCriteriaConfig = (reason: string) => {
    return REWARD_CRITERIA[reason as keyof typeof REWARD_CRITERIA] || {
      icon: Gift,
      color: "text-gray-400",
      bg: "bg-gray-800",
      border: "border-gray-700",
      label: reason,
      description: "Airdrop reward",
      category: "Other"
    };
  };

  // Группировка дропов по категориям
  const groupDropsByCategory = (drops: AirdropInfo[]) => {
    const groups: { [key: string]: AirdropInfo[] } = {};
    drops.forEach(drop => {
      const config = getCriteriaConfig(drop.reason);
      const category = config.category;
      if (!groups[category]) groups[category] = [];
      groups[category].push(drop);
    });
    return groups;
  };

  const groupedDrops = groupDropsByCategory(airdrops);
  const hasPending = parseFloat(stats.pending) > 0;

  return (
    <div className="min-h-screen bg-gradient-to-br from-[#0a0a1a] to-[#1a1a2e] p-6">
      <div className="max-w-4xl mx-auto">
        
        {/* Хедер */}
        <div className="text-center mb-8">
          <div className="flex items-center justify-center gap-3 mb-4">
            <Trophy className="w-10 h-10 text-yellow-400" />
            <h1 className="text-4xl font-bold text-white">HASH Airdrop Scanner</h1>
          </div>
          <p className="text-gray-400">Check your eligibility and claim your rewards</p>
        </div>

        {/* Подключение кошелька */}
        {!account ? (
          <button
            onClick={connectWallet}
            className="w-full mb-6 py-3 bg-blue-600 hover:bg-blue-700 text-white rounded-xl font-semibold transition"
          >
            🔌 Connect Wallet
          </button>
        ) : (
          <div className="mb-6 p-3 bg-gray-800/50 rounded-xl border border-gray-700 flex items-center justify-between">
            <div className="flex items-center gap-2">
              <div className="w-2 h-2 bg-green-400 rounded-full animate-pulse"></div>
              <span className="text-gray-300 text-sm">{account.slice(0, 6)}...{account.slice(-4)}</span>
            </div>
            <button
              onClick={() => { setAccount(null); setAirdrops([]); setStats({ total: '0', claimed: '0', pending: '0' }); }}
              className="text-gray-400 hover:text-white text-sm"
            >
              Disconnect
            </button>
          </div>
        )}

        {/* Поиск */}
        <div className="flex gap-3 mb-6">
          <input
            type="text"
            value={address}
            onChange={(e) => setAddress(e.target.value)}
            placeholder="Enter wallet address (0x...)"
            className="flex-1 px-4 py-3 bg-gray-800/50 text-white rounded-xl border border-gray-700 focus:border-blue-500 focus:outline-none transition"
          />
          <button
            onClick={() => scanAddress(address)}
            disabled={scanning || !address}
            className="px-6 py-3 bg-blue-600 hover:bg-blue-700 text-white rounded-xl font-semibold disabled:opacity-50 transition"
          >
            {scanning ? '🔍 Scanning...' : '🔍 Scan'}
          </button>
        </div>

        {/* Статистика */}
        {!loading && stats.total !== '0' && (
          <div className="grid grid-cols-3 gap-4 mb-6">
            <div className="bg-gray-800/50 p-4 rounded-xl border border-gray-700">
              <p className="text-gray-400 text-sm">Total Rewards</p>
              <p className="text-2xl font-bold text-white">{parseFloat(stats.total).toFixed(2)} HASH</p>
            </div>
            <div className="bg-gray-800/50 p-4 rounded-xl border border-green-500/30">
              <p className="text-gray-400 text-sm">Claimed</p>
              <p className="text-2xl font-bold text-green-400">{parseFloat(stats.claimed).toFixed(2)} HASH</p>
            </div>
            <div className="bg-gray-800/50 p-4 rounded-xl border border-yellow-500/30">
              <p className="text-gray-400 text-sm">Pending</p>
              <p className="text-2xl font-bold text-yellow-400">{parseFloat(stats.pending).toFixed(2)} HASH</p>
            </div>
          </div>
        )}

        {/* Кнопка Claim All */}
        {hasPending && (
          <button
            onClick={claimAll}
            className="w-full mb-6 py-3 bg-gradient-to-r from-green-600 to-blue-600 hover:opacity-90 text-white rounded-xl font-bold transition shadow-lg shadow-blue-500/20"
          >
            🎯 Claim All ({parseFloat(stats.pending).toFixed(2)} HASH)
          </button>
        )}

        {/* Список наград по категориям */}
        {!loading && airdrops.length > 0 && (
          <div className="space-y-6">
            {Object.entries(groupedDrops).map(([category, drops]) => (
              <div key={category} className="space-y-3">
                <h3 className="text-sm font-semibold text-gray-400 uppercase tracking-wider">
                  {category}
                </h3>
                {drops.map((drop, index) => {
                  const config = getCriteriaConfig(drop.reason);
                  const IconComponent = config.icon;
                  const isClaimed = drop.claimed;

                  return (
                    <div
                      key={index}
                      className={`p-4 rounded-xl border transition-all ${
                        isClaimed 
                          ? 'bg-gray-800/30 border-gray-700/50 opacity-60' 
                          : `${config.bg} ${config.border} hover:scale-[1.01]`
                      }`}
                    >
                      <div className="flex items-center gap-4">
                        {/* Иконка */}
                        <div className={`p-3 rounded-xl ${config.bg}`}>
                          <IconComponent className={`w-6 h-6 ${config.color}`} />
                        </div>

                        {/* Информация */}
                        <div className="flex-1">
                          <div className="flex items-center gap-2 flex-wrap">
                            <h4 className={`font-semibold ${config.color}`}>
                              {config.label}
                            </h4>
                            <span className="text-xs px-2 py-1 rounded-full bg-gray-700/50 text-gray-400">
                              {category}
                            </span>
                            {isClaimed && (
                              <span className="flex items-center gap-1 text-xs text-green-400">
                                <CheckCircle className="w-3 h-3" />
                                Claimed
                              </span>
                            )}
                          </div>
                          <p className="text-gray-400 text-sm">{config.description}</p>
                          <div className="flex items-center gap-4 mt-2">
                            <span className="text-white font-bold">
                              {parseFloat(drop.amount).toFixed(2)} HASH
                            </span>
                            {!isClaimed ? (
                              <span className="flex items-center gap-1 text-yellow-400 text-sm">
                                <Clock className="w-3 h-3" />
                                Pending
                              </span>
                            ) : (
                              <span className="flex items-center gap-1 text-green-400 text-sm">
                                <CheckCircle className="w-3 h-3" />
                                Received
                              </span>
                            )}
                          </div>
                        </div>

                        {/* Бейдж условия */}
                        <div className="text-right">
                          <span className="text-xs px-3 py-1 rounded-full bg-gray-700/50 text-gray-400">
                            {isClaimed ? '✅ Claimed' : '⏳ Available'}
                          </span>
                        </div>
                      </div>
                    </div>
                  );
                })}
              </div>
            ))}
          </div>
        )}

        {/* Пустое состояние */}
        {!loading && airdrops.length === 0 && stats.total === '0' && (
          <div className="text-center py-16">
            <div className="text-6xl mb-4">🔍</div>
            <p className="text-gray-400 text-lg">No airdrops found</p>
            <p className="text-gray-500 text-sm mt-2">Enter a wallet address to check eligibility</p>
          </div>
        )}

        {/* Загрузка */}
        {loading && (
          <div className="text-center py-16">
            <div className="animate-spin rounded-full h-12 w-12 border-b-2 border-blue-500 mx-auto"></div>
            <p className="text-gray-400 mt-4">Scanning for rewards...</p>
          </div>
        )}
      </div>
    </div>
  );
};

export default AirdropScanner;
